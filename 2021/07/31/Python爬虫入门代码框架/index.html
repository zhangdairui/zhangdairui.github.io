爬取单独页面文字内容

# 最简单的爬取一个单独页面的代码框架。对应固定的单一页面，可以绕过网站防爬虫设置的浏览器header检测

import requests

def getHTMLText(url):
    try:
        kv = {'user-agent':'Mozilla/5.0'}  # 把爬虫的header伪装成浏览器访问，可以绕过amazon的header审查。
        r = requests.get(url, timeout=30)  # request.get是一个request库的封装函数，其实可以用requests.request('METHOD',url,kwargs)

        # r是Response类型的简称。response类型对应5个属性可以调用，比如r.text是网页全部信息，r.status_code是爬取状态
        r.raise_for_status()  # 如果返回值不是200触发HTTP异常
        r.encoding = r.apparent_encoding  # 改变爬取的网页编码，使得print结果不是乱码
        return r.text  # 获取网页的所有信息
    except:
        return "Error!"


if __name__ == "__main__":
    url = "https://www.baidu.com/"  # 想爬取网站的地址
    print(getHTMLText(url))

向搜索引擎提交关键词并爬取搜索结果页面


# 如何使用爬虫自动向百度中提交搜索关键词，并爬取搜索结果页面

import requests
keyword = "小浣熊干脆面"

try:
    kv = {'wd' : keyword} # 这是指定了一个键值对，wd为键，意思是生成"wd=keyword"这个字符串
    # wd取决于百度网站的设定，如果是360搜索则为q，不同网站是不同的键

    r = requests.get("http://www.baidu.com/s" , params=kv)
    # params的功能是将后面的kv作为参数添加到url中。
    # 此处相当于get的url是http://www.baidu.com/s?wd=keyword(keyword一开始赋值了)

    print(r.request.url) # 把request的url输出。
    r.raise_for_status() # 判断是否为200（成功访问）
    print(len(r.text)) # 因为搜索结果太多，先把爬取的数据的大小输出看看
except:
    print("Error!")

爬取一个网络视频/图片

# 如何使用爬虫爬取固定的一张网络图片

import requests
import os # os库主要应用于管理文件夹和文件路径

# 首先设置图片的地址和爬下来存储的目录
url = "https://video.pearvideo.com/mp4/third/20200225/cont-1655810-10102052-233302-hd.mp4" # 图片的地址
root = "/Users/zhangdairui/Desktop/杂/" # 保存的目录一定要以/结尾，不然就只会存到上一级。
path = root + url.split('/')[-1]
# 上面一行意思是：最后保存的图片地址，是指定目录+图片url以斜杠分割的最后一部分，也就是说指定了文件名为网络原名

# 接下来开始爬这张图
try:
    if not os.path.exists(root): # 如果指定存储目录不存在
        os.mkdir(root) # 就新建一个目录
    if not os.path.exists(path): # 如果文件不存在，就开始执行下面的爬。
        r = requests.get(url) # get这个网页

        # 接下来开始把图片保存下来，注意图片是二进制格式
        with open(path, 'wb') as f :
            # wb的意思是写入文件。是python一种打开文件的方式。
            f.write(r.content) # r.content表示返回内容的二进制形式。将文件的二进制格式写到文件中
            f.close()
            print("保存成功！")
    else:
        print("文件已存在")
except:
    print("爬取失败")
