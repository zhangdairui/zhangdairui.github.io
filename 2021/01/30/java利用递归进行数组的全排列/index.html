java利用递归进行数组的全排列
emo | 2021年01月30日 | java 
public class Permutation {

        public static void main(String[] args) 
        {     
            int[] Array = {1,2,3};
            int n = Array.length;
            Permutation permutation = new Permutation();
            permutation.permute(Array, 0, n-1);
        } 

        private void permute(int[] arr, int l, int r)  //最开始l是0（起始位置），r是2，由main传入。
        { 
            if(l == r) { 
                for(int k=0;k<arr.length;k++) {
                    System.out.print(arr[k]);
                }
                System.out.print(" ");

            }else {
                for(int i = l; i<= r ; i++) { //最开始，i=0，l=0, i<=2 ( r=2 )
                    int temp;
                    temp = arr[i];
                    arr[i] = arr[l];
                    arr[l] = temp;

                    permute(arr, l+1, r);

                    temp = arr[i];
                    arr[i] = arr[l];
                    arr[l] = temp;
                }
            }
        }
    }
