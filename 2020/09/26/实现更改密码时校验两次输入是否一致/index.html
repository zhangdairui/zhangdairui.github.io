实现更改密码时校验两次输入是否一致

效果图：

如果两次输入密码匹配则点亮change password按钮，

同时“password do not match”字样消失。

顺便也实现了showpassword按钮，也就相当于输入密码时候的小眼睛，原理都是一样的。

代码如下：

——–HTML——–

  <div class="front-content">

    <div class="container-mid">

        <div>
          <img class="img-responsive logo" src="images/password.png" width="100px" alt="image">
        </div>
        <div class="col-sm-12">
          <h1>Change Password</h1>
        </div>
        <p class="text-center">Use the form below to change your password.</p>


        <div class="row">
          <div class="col-sm-6 col-sm-offset-3 m-auto">

              <div class="input-group mb-4 input-group-lg">
                <input type="password" class="form-control" name="password1" id="password1" placeholder="New Password" autocomplete="off">
              </div>

              <div class="input-group mb-3 input-group-lg">
                <input type="password" class="form-control" name="password1" id="password2" placeholder="Reapeat Password" autocomplete="off">
              </div>

              <div class="row">
                <div class="col-sm-12 mb-4">
                  <span id="pwmatch" class="glyphicon" style="color:red;">Passwords Do Not Match</span>
                </div>
              </div>

              <div class="form-check">
                <button class="btn btn-info btn-lg" type="button" name="showpassword" id="showpassword" value="Show Password">Show password</button>
                <input type="submit" id="ChangePassword" disabled="disabled" class="col-xs-12 btn btn-primary btn-load btn-lg" data-loading-text="Changing Password..." value="Change Password">
              </div>


        </div><!--/col-sm-6-->
      </div><!--/row-->
    </div>

——–html底部内嵌script标签——–


// Show password Button 点按按钮显示/隐藏密码
<script type="text/javascript">
$("#showpassword").on('click', function(){

  var pass1 = $("#password1");
  var pass2 = $("#password2");

  var fieldtype1 = pass1.attr('type');
  var fieldtype2 = pass2.attr('type');

  if (fieldtype1 == 'password' & fieldtype2 == 'password') {
    pass1.attr('type', 'text');
    pass2.attr('type', 'text');
    $(this).text("Hide Password");
  }else{
    pass1.attr('type', 'password');
    pass2.attr('type', 'password');
    $(this).text("Show Password");
  }


});

// Password Match 校验两次输入密码是否一致

$("input[type=password]").keyup(function(){
  if ($("#password1").val() != "" || $("#password2").val() != "") {
    if ($("#password1").val() == $("#password2").val()) {

      $("#pwmatch").css("display","none");
      $("#ChangePassword").attr('disabled',false);

    }else{
    $("#pwmatch").css("display","block");
    $("#ChangePassword").attr('disabled','disabled');
  }
}


});
//

</script>
